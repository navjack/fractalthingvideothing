   113) 	void CreateFractalBasic(const olc::vi2d& pix_tl, const olc::vi2d& pix_br, const olc::vd2d& frac_tl, const olc::vd2d& frac_br, const int iterations)
   114) 	{
   115) 		double x_scale = (frac_br.x - frac_tl.x) / (double(pix_br.x) - double(pix_tl.x));
   116) 		double y_scale = (frac_br.y - frac_tl.y) / (double(pix_br.y) - double(pix_tl.y));
   117) 
   118) 		for (int y = pix_tl.y; y < pix_br.y; y++)
   119) 		{
   120) 			for (int x = pix_tl.x; x < pix_br.x; x++)
   121) 			{
   122) 				std::complex<double> c(x * x_scale + frac_tl.x, y * y_scale + frac_tl.y);
   123) 				std::complex<double> z(0, 0);
   124) 
   125) 				int n = 0;
   126) 				while (abs(z) < 2.0 && n < iterations)
   127) 				{
   128) 					z = (z * z) + c;
   129) 					n++;
   130) 				}
   131) 
   132) 				pFractal[y * ScreenWidth() + x] = n;
   133) 			}
   134) 		}

   228) 	// Method 4) - Use AVX2 Vector co-processor to handle 4 fractal locations at once
   229) 	void CreateFractalIntrinsics(const olc::vi2d& pix_tl, const olc::vi2d& pix_br, const olc::vd2d& frac_tl, const olc::vd2d& frac_br, const int iterations)
   230) 	{
   231) 		double x_scale = (frac_br.x - frac_tl.x) / (double(pix_br.x) - double(pix_tl.x));
   232) 		double y_scale = (frac_br.y - frac_tl.y) / (double(pix_br.y) - double(pix_tl.y));
   233) 
   234) 		double y_pos = frac_tl.y;
   235) 
   236) 		int y_offset = 0;
   237) 		int row_size = ScreenWidth();
   238) 
   239) 		int x, y;
   240) 
   241) 		__m256d _a, _b, _two, _four, _mask1;
   242) 		__m256d _zr, _zi, _zr2, _zi2, _cr, _ci;
   243) 		__m256d _x_pos_offsets, _x_pos, _x_scale, _x_jump;
   244) 		__m256i _one, _c, _n, _iterations, _mask2;
   245) 
   246) 		_one = _mm256_set1_epi64x(1);
   247) 		_two = _mm256_set1_pd(2.0);
   248) 		_four = _mm256_set1_pd(4.0);
   249) 		_iterations = _mm256_set1_epi64x(iterations);
   250) 
   251) 		_x_scale = _mm256_set1_pd(x_scale);
   252) 		_x_jump = _mm256_set1_pd(x_scale * 4);
   253) 		_x_pos_offsets = _mm256_set_pd(0, 1, 2, 3);
   254) 		_x_pos_offsets = _mm256_mul_pd(_x_pos_offsets, _x_scale);
   255) 
   256) 
   257) 		for (y = pix_tl.y; y < pix_br.y; y++)
   258) 		{
   259) 			// Reset x_position
   260) 			_a = _mm256_set1_pd(frac_tl.x);
   261) 			_x_pos = _mm256_add_pd(_a, _x_pos_offsets);
   262) 
   263) 			_ci = _mm256_set1_pd(y_pos);
   264) 
   265) 			for (x = pix_tl.x; x < pix_br.x; x += 4)
   266) 			{
   267) 				_cr = _x_pos;
   268) 				_zr = _mm256_setzero_pd();
   269) 				_zi = _mm256_setzero_pd();
   270) 				_n = _mm256_setzero_si256();
   271) 
   272) 
   273) 			repeat:
   274) 				_zr2 = _mm256_mul_pd(_zr, _zr);
   275) 				_zi2 = _mm256_mul_pd(_zi, _zi);
   276) 				_a = _mm256_sub_pd(_zr2, _zi2);
   277) 				_a = _mm256_add_pd(_a, _cr);
   278) 				_b = _mm256_mul_pd(_zr, _zi);
   279) 				_b = _mm256_fmadd_pd(_b, _two, _ci);
   280) 				_zr = _a;
   281) 				_zi = _b;
   282) 				_a = _mm256_add_pd(_zr2, _zi2);
   283) 				_mask1 = _mm256_cmp_pd(_a, _four, _CMP_LT_OQ);
   284) 				_mask2 = _mm256_cmpgt_epi64(_iterations, _n);
   285) 				_mask2 = _mm256_and_si256(_mask2, _mm256_castpd_si256(_mask1));
   286) 				_c = _mm256_and_si256(_one, _mask2); // Zero out ones where n < iterations													
   287) 				_n = _mm256_add_epi64(_n, _c); // n++ Increase all n
   288) 				if (_mm256_movemask_pd(_mm256_castsi256_pd(_mask2)) > 0)
   289) 					goto repeat;
   290) 
   291) 				pFractal[y_offset + x + 0] = int(_n.m256i_i64[3]);
   292) 				pFractal[y_offset + x + 1] = int(_n.m256i_i64[2]);
   293) 				pFractal[y_offset + x + 2] = int(_n.m256i_i64[1]);
   294) 				pFractal[y_offset + x + 3] = int(_n.m256i_i64[0]);
   295) 				_x_pos = _mm256_add_pd(_x_pos, _x_jump);
   296) 			}
   297) 
   298) 			y_pos += y_scale;
   299) 			y_offset += row_size;
   300) 		}
   301) 	}
